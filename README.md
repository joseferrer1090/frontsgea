├── public
|   ├──index.html
|   ├── favicon.ico
├── src
│   ├── actions
│   ├── common
│   ├── components
│   │   ├── **/*.js
│   ├── layouts
│   ├── pages
│   ├── reducers
│   ├── store
│   ├── utils
│   ├── App.js
│   ├── index.css
│   ├── index.js
├── node_modules
├── .babelrc
├── .editorconfig
├── .gitignore
├── bower_components (if using bower)
├── package.json
├── README.md
├── bower.json (if using bower)
└── webpack.config.js
